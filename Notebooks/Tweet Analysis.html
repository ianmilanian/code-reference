Tokenize Tweets¶

We initialize word lists (positive / negative) and tokenize tweets with NLTK.

In [1]:
import nltk
import numpy as np
import pandas as pd
from matplotlib import cm
%matplotlib inline

def bold(s):
    return '\033[1m' + s + '\033[0m'

negative  = set(open("../data/negative.txt", "r").read().split())
positive  = set(open("../data/positive.txt", "r").read().split())
tweets    = [tweets for tweets in open("../data/obama_tweets.txt", "r").read().splitlines() if tweets.startswith("RT") == False]
tk_tweets = nltk.Text(nltk.TweetTokenizer().tokenize("".join(tweets).lower()))

Text Analysis¶

First we plot a simple frequency distribution for words that occured in our dataset. We also note the number of hapaxes. Next we look for words that are similar to the word "obama" and also collocations. Finally we perform a simple sentiment anaylysis based on the frequency of words that also occur in our positive / negative word lists.

tweet_fd      = nltk.FreqDist(tk_tweets)
tweet_hapaxes = tweet_fd.hapaxes()
tweet_fd.plot(15, cumulative=True)
tweet_fd.tabulate(15, cumulative=True)

print(bold("\nHapaxes:"), " ".join(tweet_hapaxes[:5]), "...")

print(bold("\nWords Near \"obama\":"))
tk_tweets.similar("obama")

print(bold("\nCollocations:"))
tk_tweets.collocations()

sentiment = {"positive": 0, "negative": 0}
for tweet in tweets:
    words = tweet.lower().split()
    sentiment["positive"] = sentiment["positive"] + len(positive.intersection(words))
    sentiment["negative"] = sentiment["negative"] + len(negative.intersection(words))

print(bold("\nTweet Sentiment:"), sentiment)

freq     = 10
df_tags  = pd.DataFrame(nltk.pos_tag(tk_tweets), columns=["word", "grammar"])
df_nouns = df_tags[(df_tags["grammar"] == "NN") & (df_tags["word"].str.len() > 1)]["word"].value_counts()
df_nouns = df_nouns[df_nouns[:,] > freq]
df_nouns.plot.pie(title="Tweeted Nouns With %d+ Occurances" % freq, figsize=(12,12), colormap=cm.get_cmap('viridis'))
