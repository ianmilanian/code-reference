from sklearn.linear_model import LinearRegression

def kfold_test(data, num_folds, x, y):
    accuracies    = []
    kregr         = LinearRegression()    
    shuffled_data = data.loc[np.random.permutation(len(data))].reset_index(drop=True)
    split         = int(len(shuffled_data) / num_folds)
    
    for i in range(num_folds):
        test  = shuffled_data[i*split:][:split]
        train = pd.concat([shuffled_data[:i*split], shuffled_data[(i+1)*split:]])
        kregr.fit(train[[x]], train[y])
        correct_predictions = test[[round(val) for val in kregr.predict(test[[x]])] == round(test[y])]
        accuracies.append(len(correct_predictions) / len(test))

    return np.mean(accuracies)

from sklearn.cluster import KMeans

def calculate_centroids(data, x, y, n):
    centroid_dict = {}
    if "cluster" not in data:
        #Calculate Initial Centroids
        count            = 0
        random_centroids = data.loc[np.random.choice(data.index, size=n)]
        for i, centroid in random_centroids.iterrows():
            centroid_dict[count] = [centroid[x], centroid[y]]
            count += 1
    else:
        #Recalculate Centroids
        for cluster_id in range(0, n):
            clustered_data            = data[data["cluster"] == cluster_id]
            clustered_coords          = [clustered_data[x].mean(), clustered_data[y].mean()]
            centroid_dict[cluster_id] = clustered_coords
        
    return centroid_dict

def assign_cluster(row, x, y, centroids):
    curr_closest = -1
    cluster_id   = -1
    target       = [row[x], row[y]]
    for i, centroid in centroids.items():
        euclidean_distance = sum([(centroid[i] - target[i])**2 for i in range(0, len(target))])**0.5
        if  cluster_id == -1 or euclidean_distance < curr_closest:
            cluster_id   = i
            curr_closest = euclidean_distance

    return cluster_id

def kmeans_clustering(df, x, y, n, itr):
    
    km = KMeans(n, random_state=0).fit(auto[[x, y]])
    
    #TODO: NORMALIZE x and y!! Euclidean Distance will skew results otherwise!
    # (x - xmean) / xstd
    
    step  = 0;
    data  = df.copy() #don't modify original dataframe
    axarr = plt.subplots(itr, itr)[1] #unpack ax array
    for i, axlist in enumerate(axarr):
        for j, ax in enumerate(axlist):
            ax.grid('off')
            ax.set_xticks([])
            ax.set_yticks([])         
            centroids       = calculate_centroids(data, x, y, n)
            data["cluster"] = data.apply(lambda row: assign_cluster(row, x, y, centroids), axis=1)
            colors          = ['b', 'g', 'c', 'm', 'y', 'k', 'b', 'g', 'c', 'm', 'y', 'k']
            scale_data      =  25/itr            
            scale_centroid  =  75/itr
            scale_influence = 100/itr * data["cluster"].value_counts().sort_index()
            centroid_x, centroid_y = zip(*centroids.values())
            step += 1
            if i == len(axarr)-1 and j == len(axlist)-1:
                ax.set_title("Scikit KMeans", x=0.5, y=0.85)
                ax.scatter(data[x], data[y], c=[colors[label] for label in km.labels_], s=scale_data)
            else:
                ax.set_title("Step = %d" % step, x=0.5, y=0.85)
                for k in range(len(centroids.items())):
                    clustered_data = data[data["cluster"] == k]
                    ax.scatter(centroid_x[k],     centroid_y[k],     c=colors[k], s=scale_influence[k], alpha=0.1) 
                    ax.scatter(clustered_data[x], clustered_data[y], c=colors[k], s=scale_data)
                    ax.scatter(centroid_x[k],     centroid_y[k],     c="r",       s=scale_centroid)

    plt.tight_layout()
    plt.show()

def multiclass_classification(data):
    data = pd.concat([data, pd.get_dummies(data["cylinders"], prefix="cyl")],  axis=1)

    shuffled_data  = data.loc[np.random.permutation(len(data))].reset_index(drop=True)
